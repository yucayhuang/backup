	public string url;
 
	private RawImage rawImage;
 
	void Start () {
		rawImage = this.GetComponent<RawImage> ();
		StartCoroutine (LoadImage());
	}
	
	IEnumerator LoadImage()
 
	{
		WWW www = new WWW (url);
		yield return www;
